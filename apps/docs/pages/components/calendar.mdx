# Calendar

A date picker component that allows users to select dates from a calendar interface.

## Installation

```bash
npm install @ui/components
```

## Import

```tsx
import { Calendar } from '@ui/components';
```

## Usage

```tsx
export default function CalendarDemo() {
  const [date, setDate] = useState<Date>();

  return (
    <Calendar mode="single" selected={date} onSelect={setDate} className="rounded-md border" />
  );
}
```

## Features

- Single date selection
- Multiple date selection
- Date range selection
- Customizable styling
- Keyboard navigation
- Localization support
- Disabled dates

## Props

### Root

| Prop              | Type                                 | Description                         |
| ----------------- | ------------------------------------ | ----------------------------------- |
| `mode`            | `'single' \| 'multiple' \| 'range'`  | Selection mode                      |
| `selected`        | `Date \| Date[] \| DateRange`        | Selected date(s)                    |
| `onSelect`        | `(date: Date) => void`               | Selection change callback           |
| `disabled`        | `Date[] \| { from: Date; to: Date }` | Disabled date(s)                    |
| `initialFocus`    | `boolean`                            | Whether to focus on mount           |
| `weekStartsOn`    | `0 \| 1 \| 2 \| 3 \| 4 \| 5 \| 6`    | First day of the week               |
| `locale`          | `Locale`                             | Calendar localization               |
| `className`       | `string`                             | Optional class name for styling     |
| `showOutsideDays` | `boolean`                            | Show dates from previous/next month |

## Examples

### Single Date Selection

```tsx
<Calendar mode="single" selected={date} onSelect={setDate} className="rounded-md border" />
```

### Date Range Selection

```tsx
const [range, setRange] = useState<DateRange>({
  from: new Date(),
  to: addDays(new Date(), 7),
});

<Calendar
  mode="range"
  selected={range}
  onSelect={setRange}
  numberOfMonths={2}
  className="rounded-md border"
/>;
```

### With Disabled Dates

```tsx
<Calendar
  mode="single"
  selected={date}
  onSelect={setDate}
  disabled={{
    from: new Date(),
    to: addDays(new Date(), 7),
  }}
  className="rounded-md border"
/>
```

### Custom Locale

```tsx
import { fr } from 'date-fns/locale';

<Calendar
  mode="single"
  selected={date}
  onSelect={setDate}
  locale={fr}
  className="rounded-md border"
/>;
```
