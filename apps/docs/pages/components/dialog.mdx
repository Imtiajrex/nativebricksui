# Dialog

A window overlaid on either the primary window or another dialog window, rendering the content underneath inert.

## Installation

```bash
npm install @ui/components
```

## Import

```tsx
import { Dialog } from '@ui/components';
```

## Usage

```tsx
export default function DialogDemo() {
  const [open, setOpen] = useState(false);

  return (
    <>
      <Button onClick={() => setOpen(true)}>Open Dialog</Button>
      <Dialog
        open={open}
        onOpenChange={setOpen}
        title="Edit Profile"
        description="Make changes to your profile here."
        Footer={
          <div className="flex justify-end gap-3">
            <Button variant="outline" onClick={() => setOpen(false)}>
              Cancel
            </Button>
            <Button onClick={() => setOpen(false)}>Save changes</Button>
          </div>
        }
      >
        <div className="p-4 space-y-4">
          <Input label="Name" placeholder="Enter your name" />
          <Input label="Email" type="email" placeholder="Enter your email" />
        </div>
      </Dialog>
    </>
  );
}
```

## Features

- Modal dialog with overlay
- Focus management
- Keyboard navigation (Esc to close)
- Customizable header and footer
- Accessible ARIA dialog
- Mobile responsive
- Imperative API support

## Props

### Root

| Prop              | Type                      | Description                      |
| ----------------- | ------------------------- | -------------------------------- |
| `open`            | `boolean`                 | Whether the dialog is open       |
| `onOpenChange`    | `(open: boolean) => void` | Callback when open state changes |
| `title`           | `string`                  | Optional dialog title            |
| `description`     | `string`                  | Optional dialog description      |
| `children`        | `React.ReactNode`         | Dialog content                   |
| `Header`          | `React.ReactNode`         | Custom header component          |
| `Footer`          | `React.ReactNode`         | Custom footer component          |
| `className`       | `string`                  | Optional class name for styling  |
| `headerClassName` | `string`                  | Class name for header styling    |
| `footerClassName` | `string`                  | Class name for footer styling    |

### Imperative API

```tsx
function DialogWithImperativeAPI() {
  const dialogRef = useRef<Dialog>(null);

  const handleOpenDialog = () => {
    dialogRef.current?.show();
  };

  const handleCloseDialog = () => {
    dialogRef.current?.hide();
  };

  return (
    <>
      <Button onClick={handleOpenDialog}>Open Dialog</Button>
      <Dialog
        ref={dialogRef}
        title="Imperative Dialog"
        description="This dialog is controlled imperatively."
      >
        <div className="p-4">
          <p>Dialog content here.</p>
          <Button onClick={handleCloseDialog}>Close</Button>
        </div>
      </Dialog>
    </>
  );
}
```
